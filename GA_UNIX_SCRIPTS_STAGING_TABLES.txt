CREATE TABLE ga_stg_sales (
ORDER_NUMBER VARCHAR(10) NOT NULL,
PROD_CODE VARCHAR(10) NOT NULL,
CUST_CODE VARCHAR(10) NOT NULL,
SALE_DATE DATE,
CHANNEL_CODE VARCHAR(10) NOT NULL,
PROMO_CODE VARCHAR(10) NOT NULL,
QUANTITY_SOLD VARCHAR(10),

LAST_UPDATE_DT DATE,
STG_LOAD_DATE DATE NOT NULL);
